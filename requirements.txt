# ── Chrome driver (persistent) ────────────────────────────────────────────────
def get_driver():
    global DRIVER
    with DRIVER_LOCK:
        if DRIVER:
            try:
                _ = DRIVER.title  # alive check
                return DRIVER
            except:
                DRIVER = None
        if not HAS_SELENIUM:
            return None
        try:
            opts = Options()
            opts.add_argument("--start-maximized")
            opts.add_argument("--disable-infobars")
            opts.add_experimental_option("excludeSwitches", ["enable-automation"])
            opts.add_experimental_option("useAutomationExtension", False)
            opts.add_argument("--disable-blink-features=AutomationControlled")
            # Use existing Chrome profile so you're logged in to YouTube
            profile = _chrome_profile_path()
            if profile:
                opts.add_argument(f"--user-data-dir={profile}")
            DRIVER = webdriver.Chrome(options=opts)
            return DRIVER
        except Exception as e:
            print(f"[AVA] Chrome driver error: {e}")
            return None

def _chrome_profile_path():
    """Return Chrome's default profile path for the OS."""
    home = Path.home()
    if OS == "Windows":
        p = home / "AppData/Local/Google/Chrome/User Data"
    elif OS == "Darwin":
        p = home / "Library/Application Support/Google/Chrome"
    elif OS == "Linux":
        p = home / ".config/google-chrome"
    else:
        return None
    return str(p) if p.exists() else None

# ── YouTube control ───────────────────────────────────────────────────────────
def youtube_play(query: str) -> dict:
    """Search YouTube and auto-play the first video result."""
    driver = get_driver()
    if not driver:
        # Fallback: open in default browser
        url = f"https://www.youtube.com/results?search_query={quote_plus(query)}&autoplay=1"
        webbrowser.open(url)
        return {"success": True, "method": "browser_fallback", "message": f"Opened YouTube search for: {query}"}

    try:
        search_url = f"https://www.youtube.com/results?search_query={quote_plus(query)}"
        driver.get(search_url)
        time.sleep(2)

        # Click first video result
        wait = WebDriverWait(driver, 10)
        video = wait.until(EC.element_to_be_clickable(
            (By.CSS_SELECTOR, "ytd-video-renderer a#video-title, ytd-compact-video-renderer a#video-title")
        ))
        title = video.get_attribute("title") or query
        driver.execute_script("arguments[0].click();", video)
        time.sleep(2)

        # Unmute and max volume
        try:
            driver.execute_script("""
                var v = document.querySelector('video');
                if(v){ v.muted=false; v.volume=1; }
            """)
        except: pass

        return {"success": True, "method": "selenium", "playing": title, "message": f"Now playing: {title}"}

    except Exception as e:
        webbrowser.open(f"https://www.youtube.com/results?search_query={quote_plus(query)}")
        return {"success": True, "method": "browser_fallback", "message": f"Opened YouTube for: {query}"}

def youtube_control(action: str) -> dict:
    """Control YouTube playback: pause, resume, next, prev, volume."""
    driver = get_driver()
    if not driver:
        return {"success": False, "message": "Chrome not running"}

    try:
        scripts = {
            "pause":      "document.querySelector('video').pause();",
            "play":       "document.querySelector('video').play();",
            "mute":       "document.querySelector('video').muted=true;",
            "unmute":     "document.querySelector('video').muted=false;",
            "vol_up":     "var v=document.querySelector('video');v.volume=Math.min(1,v.volume+0.2);",
            "vol_down":   "var v=document.querySelector('video');v.volume=Math.max(0,v.volume-0.2);",
            "fullscreen": "document.querySelector('.ytp-fullscreen-button').click();",
            "like":       "document.querySelector('like-button-view-model button, yt-icon-button#like-button button').click();",
        }
        # Next / prev via keyboard
        if action == "next":
            driver.find_element(By.TAG_NAME, "body").send_keys("N")
        elif action == "prev":
            driver.find_element(By.TAG_NAME, "body").send_keys("P")
        elif action in scripts:
            driver.execute_script(scripts[action])
        else:
            return {"success": False, "message": f"Unknown action: {action}"}

        return {"success": True, "action": action}
    except Exception as e:
        return {"success": False, "message": str(e)}

@app.get("/stats")
async def get_stats():
    data = {"os": OS, "time": datetime.datetime.now().isoformat(), "selenium": HAS_SELENIUM}
    if HAS_PSUTIL:
        bat = psutil.sensors_battery()
        data.update({
            "cpu":          round(psutil.cpu_percent(interval=0.5), 1),
            "ram":          round(psutil.virtual_memory().percent, 1),
            "ram_used_gb":  round(psutil.virtual_memory().used / 1e9, 2),
            "ram_total_gb": round(psutil.virtual_memory().total / 1e9, 2),
            "disk":         round(psutil.disk_usage('/').percent, 1),
            "battery":      round(bat.percent) if bat else None,
            "plugged":      bat.power_plugged if bat else None,
            "cores":        psutil.cpu_count(),
        })
    return JSONResponse(data)

@app.post("/youtube")
async def yt_endpoint(payload: dict):
    action = payload.get("action", "play")   # play | pause | resume | next | prev | vol_up | vol_down | mute | unmute
    query  = payload.get("query", "")

    if action == "play" and query:
        result = await asyncio.to_thread(youtube_play, query)
    else:
        result = await asyncio.to_thread(youtube_control, action)
    return JSONResponse(result)

@app.post("/execute")
async def execute(payload: dict):
    cmd = payload.get("command", "").lower().strip()
    result = {"success": False, "message": "Not recognized", "action": "unknown"}

    # Local app
    for name, cmds in APP_MAP.items():
        if name in cmd:
            shell_cmd = cmds.get(OS, "")
            if shell_cmd:
                try:
                    subprocess.Popen(shell_cmd, shell=True)
                    result = {"success": True, "message": f"Launching {name}", "action": "app_launch"}
                    return JSONResponse(result)
                except Exception as e:
                    result = {"success": False, "message": str(e)}
                    return JSONResponse(result)

    # Website
    for site, url in WEB_MAP.items():
        if site in cmd:
            driver = get_driver()
            if driver:
                await asyncio.to_thread(driver.get, url)
            else:
                webbrowser.open(url)
            return JSONResponse({"success": True, "message": f"Opening {url}", "action": "web_open"})

    # Web search
    if any(w in cmd for w in ["search for","search ","look up","find info"]):
        query = cmd
        for w in ["search for","search","look up","find info"]:
            query = query.replace(w, "").strip()
        driver = get_driver()
        url = f"https://google.com/search?q={quote_plus(query)}"
        if driver:
            await asyncio.to_thread(driver.get, url)
        else:
            webbrowser.open(url)
        return JSONResponse({"success": True, "message": f"Searching: {query}", "action": "web_search"})

    # Volume
    if "volume up" in cmd:   _vol("up");   return JSONResponse({"success":True,"action":"volume"})
    if "volume down" in cmd: _vol("down"); return JSONResponse({"success":True,"action":"volume"})
    if "mute" in cmd:        _vol("mute"); return JSONResponse({"success":True,"action":"mute"})

    # Screenshot
    if "screenshot" in cmd:
        path = await asyncio.to_thread(_screenshot)
        return JSONResponse({"success": True, "message": f"Saved: {path}", "action": "screenshot"})

    # Lock screen
    if "lock" in cmd:
        _lock()
        return JSONResponse({"success": True, "action": "lock"})

    return JSONResponse(result)

@app.websocket("/ws/stats")
async def stats_ws(ws: WebSocket):
    await ws.accept()
    try:
        while True:

                bat = psutil.sensors_battery()
                data = {
                    "cpu":     round(psutil.cpu_percent(), 1),

            await asyncio.sleep(2)
    except WebSocketDisconnect:
        pass

# ── System helpers ────────────────────────────────────────────────────────────
def _vol(action):
    if OS == "Windows":
        from ctypes import cast, POINTER
        try:
            from comtypes import CLSCTX_ALL
            from pycaw.pycaw import AudioUtilities, IAudioEndpointVolume
            devices = AudioUtilities.GetSpeakers()
            interface = devices.Activate(IAudioEndpointVolume._iid_, CLSCTX_ALL, None)
            volume = cast(interface, POINTER(IAudioEndpointVolume))
            cur = volume.GetMasterVolumeLevelScalar()
            if action == "up":   volume.SetMasterVolumeLevelScalar(min(1.0, cur+0.1), None)
            elif action == "down": volume.SetMasterVolumeLevelScalar(max(0.0, cur-0.1), None)
            elif action == "mute": volume.SetMute(1, None)
        except:
            # Fallback: nircmd
            if action == "up":   subprocess.run(["nircmd","changesysvolume","5000"],capture_output=True)
            elif action == "down": subprocess.run(["nircmd","changesysvolume","-5000"],capture_output=True)
    elif OS == "Darwin":
        levels = {"up":"(output volume of (get volume settings) + 10)","down":"(output volume of (get volume settings) - 10)"}
        if action in levels:
            subprocess.run(["osascript","-e",f"set volume output volume {levels[action]}"])
        elif action == "mute":
            subprocess.run(["osascript","-e","set volume with output muted"])
    elif OS == "Linux":
        cmds = {"up":["amixer","-D","pulse","sset","Master","10%+"],"down":["amixer","-D","pulse","sset","Master","10%-"],"mute":["amixer","-D","pulse","sset","Master","toggle"]}
        if action in cmds: subprocess.run(cmds[action])

def _screenshot():
    path = str(Path.home() / f"AVA_shot_{int(time.time())}.png")
    if OS == "Darwin":   subprocess.run(["screencapture", path])
    elif OS == "Linux":  subprocess.run(["scrot", path])
    elif OS == "Windows":
        try:
            import pyautogui; pyautogui.screenshot().save(path)
        except: path = "Install pyautogui for screenshots"
    return path

def _lock():
    if OS == "Windows":  subprocess.run(["rundll32","user32.dll,LockWorkStation"])
    elif OS == "Darwin": subprocess.run(["/System/Library/CoreServices/Menu Extras/User.menu/Contents/Resources/CGSession","-suspend"])
    elif OS == "Linux":  subprocess.run(["gnome-screensaver-command","--lock"])
